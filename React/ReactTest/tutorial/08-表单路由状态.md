## 8.1 表单测试

### 受控表单基础

表单测试的核心是：**模拟用户填写 → 提交 → 验证结果**。

```typescript
// LoginForm.tsx
interface LoginFormProps {
  onSubmit: (data: { email: string; password: string }) => Promise<void>
}

function LoginForm({ onSubmit }: LoginFormProps) {
  const [email, setEmail] = useState('')
  const [password, setPassword] = useState('')
  const [error, setError] = useState('')
  const [isSubmitting, setIsSubmitting] = useState(false)

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault()
    setError('')

    if (!email.includes('@')) {
      setError('请输入有效的邮箱地址')
      return
    }
    if (password.length < 6) {
      setError('密码至少 6 个字符')
      return
    }

    setIsSubmitting(true)
    try {
      await onSubmit({ email, password })
    } catch (err) {
      setError(err instanceof Error ? err.message : '登录失败')
    } finally {
      setIsSubmitting(false)
    }
  }

  return (
    <form onSubmit={handleSubmit}>
      <div>
        <label htmlFor="email">邮箱</label>
        <input
          id="email"
          type="email"
          value={email}
          onChange={(e) => setEmail(e.target.value)}
        />
      </div>
      <div>
        <label htmlFor="password">密码</label>
        <input
          id="password"
          type="password"
          value={password}
          onChange={(e) => setPassword(e.target.value)}
        />
      </div>
      {error && <p role="alert">{error}</p>}
      <button type="submit" disabled={isSubmitting}>
        {isSubmitting ? '登录中...' : '登录'}
      </button>
    </form>
  )
}
```

// LoginForm.test.tsx

  it('提交过程中按钮显示 loading 并被禁用', async () => {
    const user = userEvent.setup()

      () => new Promise((resolve) => setTimeout(resolve, 1000))
    )

    render(<LoginForm onSubmit={onSubmit} />)

    await user.type(screen.getByLabelText('邮箱'), 'alice@example.com')
    await user.type(screen.getByLabelText('密码'), 'password123')
    await user.click(screen.getByRole('button'))

    // 提交后立即检查
    expect(screen.getByRole('button')).toHaveTextContent('登录中...')
    expect(screen.getByRole('button')).toBeDisabled()
  })
})

### React Hook Form + Zod 验证

```typescript
// SignUpForm.tsx
import { useForm } from 'react-hook-form'
import { z } from 'zod'
import { zodResolver } from '@hookform/resolvers/zod'

const signUpSchema = z.object({
  email: z.string().email('请输入有效的邮箱'),
  password: z.string().min(8, '密码至少 8 位'),
  confirmPassword: z.string(),
}).refine((data) => data.password === data.confirmPassword, {
  message: '两次密码不一致',
  path: ['confirmPassword'],
})

type SignUpData = z.infer<typeof signUpSchema>

function SignUpForm({ onSubmit }: { onSubmit: (data: SignUpData) => Promise<void> }) {
  const { register, handleSubmit, formState: { errors, isSubmitting } } = useForm<SignUpData>({
    resolver: zodResolver(signUpSchema),
  })

  return (
    <form onSubmit={handleSubmit(onSubmit)}>
      <div>
        <label htmlFor="email">邮箱</label>
        <input id="email" {...register('email')} />
        {errors.email && <p role="alert">{errors.email.message}</p>}
      </div>
      <div>
        <label htmlFor="password">密码</label>
        <input id="password" type="password" {...register('password')} />
        {errors.password && <p role="alert">{errors.password.message}</p>}
      </div>
      <div>
        <label htmlFor="confirmPassword">确认密码</label>
        <input id="confirmPassword" type="password" {...register('confirmPassword')} />
        {errors.confirmPassword && <p role="alert">{errors.confirmPassword.message}</p>}
      </div>
      <button type="submit" disabled={isSubmitting}>注册</button>
    </form>
  )
}
```

```typescript
describe('SignUpForm', () => {
  it('提交空表单时显示所有必填字段错误', async () => {
    const user = userEvent.setup()
    const onSubmit = vi.fn()

    render(<SignUpForm onSubmit={onSubmit} />)

    await user.click(screen.getByRole('button', { name: '注册' }))

    // React Hook Form + Zod 触发了所有验证错误
    const alerts = await screen.findAllByRole('alert')
    expect(alerts).toHaveLength(3) // email + password + confirm
    expect(onSubmit).not.toHaveBeenCalled()
  })

  it('密码不匹配时显示自定义验证消息', async () => {
    const user = userEvent.setup()
    const onSubmit = vi.fn()

    render(<SignUpForm onSubmit={onSubmit} />)

    await user.type(screen.getByLabelText('邮箱'), 'test@example.com')
    await user.type(screen.getByLabelText('密码'), 'password123')
    await user.type(screen.getByLabelText('确认密码'), 'different')
    await user.click(screen.getByRole('button', { name: '注册' }))

    expect(await screen.findByText('两次密码不一致')).toBeInTheDocument()
    expect(onSubmit).not.toHaveBeenCalled()
  })
})
```

> **自我验证说明**：`react-hook-form` 的 `register` 返回 `{ onChange, onBlur, name, ref }`，直接展开赋值给 `<input>`。`@hookform/resolvers/zod` 的 `zodResolver` 将 Zod schema 转为 RHF 的验证函数。`findAllByRole` 查询所有匹配的元素。

### Zustand

```typescript
// store.ts
import { create } from 'zustand'

interface BearStore {
  bears: number
  increase: () => void
  reset: () => void
}

export const useBearStore = create<BearStore>((set) => ({
  bears: 0,
  increase: () => set((state) => ({ bears: state.bears + 1 })),
  reset: () => set({ bears: 0 }),
}))
```

```typescript
// BearCounter.test.tsx
import { useBearStore } from './store'

describe('BearCounter', () => {
  // 每个测试前重置 store
  beforeEach(() => {
    useBearStore.setState({ bears: 0 })
  })

  it('显示初始值为 0', () => {
    render(<BearCounter />)
    expect(screen.getByText('Bears: 0')).toBeInTheDocument()
  })

  it('点击按钮增加数量', async () => {
    const user = userEvent.setup()
    render(<BearCounter />)

    await user.click(screen.getByRole('button', { name: /增加/i }))

    expect(screen.getByText('Bears: 1')).toBeInTheDocument()
  })
})
```

> **自我验证说明**：Zustand 的 `setState` 可以直接在测试中修改 store 状态，不需要 mock。在 `beforeEach` 中重置确保了测试隔离。这个模式比 `vi.mock('zustand')` 更简单且更接近真实行为。

### Redux Toolkit

```typescript
// test-utils 扩展
import { configureStore } from '@reduxjs/toolkit'
import { Provider } from 'react-redux'
import { rootReducer, RootState } from '@/store'

export function renderWithStore(
  ui: ReactElement,
  { preloadedState }: { preloadedState?: Partial<RootState> } = {}
) {
  const store = configureStore({
    reducer: rootReducer,
    preloadedState: preloadedState as RootState,
  })

  function Wrapper({ children }: { children: React.ReactNode }) {
    return <Provider store={store}>{children}</Provider>
  }

  return {
    store,
    ...render(ui, { wrapper: Wrapper }),
  }
}

// 使用
it('渲染用户从 store 中获取的数据', () => {
  renderWithStore(<UserDisplay />, {
    preloadedState: {
      user: { id: '1', name: 'Alice', email: 'alice@example.com' },
    },
  })

  expect(screen.getByText('Alice')).toBeInTheDocument()
})
```

describe('ProductSearch — 集成测试', () => {
  it('搜索 → 显示结果 → 添加到购物车', async () => {
    const user = userEvent.setup()

    // 设置 MSW handler
    server.use(
      http.get('/api/products', ({ request }) => {
        const url = new URL(request.url)
        const query = url.searchParams.get('q') ?? ''
        if (query === 'laptop') {
          return HttpResponse.json([
            { id: 1, name: 'Gaming Laptop', price: 999 },
            { id: 2, name: 'Ultrabook', price: 799 },
          ])
        }
        return HttpResponse.json([])
      })
    )

    renderWithProviders(<ProductSearch />)

    // 1. 初始状态：搜索框 + 空状态
    expect(screen.getByRole('textbox', { name: /搜索商品/i })).toBeInTheDocument()
    expect(screen.getByText(/输入关键词开始搜索/i)).toBeInTheDocument()

    // 2. 输入搜索词
    await user.type(screen.getByRole('textbox', { name: /搜索商品/i }), 'laptop')
    await user.click(screen.getByRole('button', { name: /搜索/i }))

    // 3. 验证结果
    expect(await screen.findByText('Gaming Laptop')).toBeInTheDocument()
    expect(screen.getByText('$999')).toBeInTheDocument()
    expect(screen.getByText('Ultrabook')).toBeInTheDocument()

    // 4. 添加第一个商品到购物车
    const addButtons = screen.getAllByRole('button', { name: /加入购物车/i })
    await user.click(addButtons[0])

    // 5. 验证购物车更新
    expect(screen.getByTestId('cart-count')).toHaveTextContent('1')
  })

  it('搜索无结果时显示空状态', async () => {
    const user = userEvent.setup()

    renderWithProviders(<ProductSearch />)

    await user.type(screen.getByRole('textbox', { name: /搜索商品/i }), 'qwertyuiop')
    await user.click(screen.getByRole('button', { name: /搜索/i }))

    expect(await screen.findByText(/未找到相关商品/i)).toBeInTheDocument()
  })
})

## 本章总结

- 表单测试 = 用户填写 → 验证 → 提交 → 验证结果
- React Router 测试用 `MemoryRouter` 替代真实浏览器路由
- Context 测试：直接在 Provider 中注入模拟值
- Zustand 测试：用 `setState` 控制状态 + `beforeEach` 重置
- Redux 测试：用 `configureStore` + `preloadedState` 注入初始状态
- 复合场景测试：串联多个用户步骤，验证完整业务流程
